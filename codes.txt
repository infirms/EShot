void CaptureOverlay::startCapture() { m_isSelecting = false; m_selectionComplete = false; m_selectionStart = QPoint(); m_selectionEnd = QPoint();

    m_annotationEngine->clear();
}

void CaptureOverlay::captureAllScreens() { // 1. Calculate Virtual Desktop Rect in LOGICAL coordinates QRect virtualRect; QList<QScreen*> screens = QGuiApplication::screens();

for (QScreen *screen : screens) {
    virtualRect = virtualRect.united(screen->geometry());
}
m_virtualDesktopRect = virtualRect;
// 2. Create composite pixmap (LOGICAL size)
// Qt handles High-DPI scaling internally for QPixmap.
m_screenSnapshot = QPixmap(m_virtualDesktopRect.size());
m_screenSnapshot.fill(Qt::black);
// 3. Draw each screen
QPainter painter(&m_screenSnapshot);
for (QScreen *screen : screens) {
    // grabWindow(0) returns native pixels.
    QPixmap screenGrab = screen->grabWindow(0);
    qreal dpr = screen->devicePixelRatio();
    
    // geometry() tells us the logical position.
    QRect screenGeo = screen->geometry();
    QPoint targetPos = screenGeo.topLeft() - m_virtualDesktopRect.topLeft();
    
    // IMPORTANT: We must specify the logical size to draw into.
    // If we just say drawPixmap(pos, pix), Qt uses pix.size() / pix.devicePixelRatio().
    // But screenGrab comes from grabWindow, which might not have the DPR set correctly
    // (it might just be a raw bitmap).
    // So we explicitly calculate the logical size: Native / DPR.
    
    int logicalW = screenGrab.width() / dpr;
    int logicalH = screenGrab.height() / dpr;
    
    painter.drawPixmap(targetPos.x(), targetPos.y(), logicalW, logicalH, screenGrab);
    
    qDebug() << "[CaptureOverlay] Screen:" << screen->name() 
             << " Geo:" << screenGeo << " DPR:" << dpr
             << " Native:" << screenGrab.size() 
             << " Draw:" << logicalW << "x" << logicalH;
}
painter.end();
}

void CaptureOverlay::paintEvent(QPaintEvent *event) { Q_UNUSED(event);

QPainter painter(this);
// Draw the captured screen background
painter.drawPixmap(0, 0, m_screenSnapshot);
// Dim the background
painter.fillRect(rect(), QColor(0, 0, 0, 100));
// Calculate selection rect
QRect selRect = normalizedSelectionRect();
if (!selRect.isEmpty()) {
    // Draw clear selected area (source composite)
    painter.setCompositionMode(QPainter::CompositionMode_Source);
    painter.drawPixmap(selRect, m_screenSnapshot, selRect);
    
    // Go back to drawing over
    painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
    
    // Draw border
    QPen borderPen(QColor(0, 122, 204), 2);
    painter.setPen(borderPen);
    painter.setBrush(Qt::NoBrush);
    painter.drawRect(selRect);
    
    // Draw annotations clipped to selection
    if (m_annotationEngine && m_selectionComplete) {
        painter.setClipRect(selRect);
        m_annotationEngine->render(&painter, selRect.topLeft());
        painter.setClipping(false);
    }
    
    // Draw dimensions text
    if (m_isSelecting) {
        QString dimensions = QString("%1 x %2").arg(selRect.width()).arg(selRect.height());
         int labelX = selRect.left();
         int labelY = selRect.top() - 25;
         if (labelY < 0) labelY = selRect.top() + 5;
         
         painter.setPen(Qt::white);
         painter.drawText(labelX, labelY, dimensions);
    }
}
}

void CaptureOverlay::mousePressEvent(QMouseEvent *event) { if (event->button() == Qt::LeftButton) { if (m_toolbar && m_toolbar->isVisible() && m_toolbar->geometry().contains(event->pos())) { return; }

if (m_selectionComplete) {
        QRect selRect = normalizedSelectionRect();
        if (selRect.contains(event->pos())) {
            if (m_annotationEngine) {
                 m_annotationEngine->beginDraw(event->pos() - selRect.topLeft());
                 update();
            }
        } else {
            m_selectionComplete = false;
            m_isSelecting = true;
            m_selectionStart = event->pos();
            m_selectionEnd = event->pos();
            hideToolbar();
            if (m_annotationEngine) m_annotationEngine->clear();
            update();
        }
    } else {
        m_isSelecting = true;
        m_selectionStart = event->pos();
        m_selectionEnd = event->pos();
        hideToolbar();
        update();
    }
}
}

void CaptureOverlay::mouseMoveEvent(QMouseEvent *event) { if (m_isSelecting) { m_selectionEnd = event->pos(); update(); } else if (m_selectionComplete && m_annotationEngine && m_annotationEngine->currentTool() != AnnotationEngine::None) { QRect selRect = normalizedSelectionRect(); if (selRect.contains(event->pos())) { m_annotationEngine->continueDraw(event->pos() - selRect.topLeft()); update(); } } }

void CaptureOverlay::mouseReleaseEvent(QMouseEvent *event) { if (event->button() == Qt::LeftButton) { if (m_isSelecting) { m_isSelecting = false; m_selectionEnd = event->pos();

QRect selRect = normalizedSelectionRect();
        if (selRect.width() > 10 && selRect.height() > 10) {
            m_selectionComplete = true;
            showToolbar();
        }
        update();
    } else if (m_selectionComplete) {
        QRect selRect = normalizedSelectionRect();
        QPoint relativePos = event->pos() - selRect.topLeft();
        
        relativePos.setX(qBound(0, relativePos.x(), selRect.width()));
        relativePos.setY(qBound(0, relativePos.y(), selRect.height()));
        
        if (m_annotationEngine && m_annotationEngine->currentTool() == AnnotationEngine::Text) {
            if (selRect.contains(event->pos())) {
                bool ok;
                QString text = QInputDialog::getText(this, tr("Metin Ekle"), tr("Yazı:"), QLineEdit::Normal, "", &ok);
                if (ok && !text.isEmpty()) {
                    m_annotationEngine->addTextAnnotation(relativePos, text);
                    update();
                }
            }
        } else if (m_annotationEngine) {
            m_annotationEngine->endDraw(relativePos);
            update();
        }
    }
}
}

void CaptureOverlay::keyPressEvent(QKeyEvent *event) { if (event->key() == Qt::Key_Escape) { if (m_selectionComplete) { m_selectionComplete = false; m_isSelecting = false; m_selectionStart = QPoint(); hideToolbar(); if (m_annotationEngine) m_annotationEngine->clear(); update(); } else { onClose(); } } else if (event->matches(QKeySequence::Copy)) { onCopyToClipboard(); } else if (event->matches(QKeySequence::Save)) { onSave(); } else if (event->matches(QKeySequence::Undo)) { if (m_annotationEngine) { m_annotationEngine->undo(); update(); } } else if (event->matches(QKeySequence::Redo)) { if (m_annotationEngine) { m_annotationEngine->redo(); update(); } } }

QRect selRect = normalizedSelectionRect();
int x = selRect.right() - m_toolbar->width();
int y = selRect.bottom() + 10;
if (x < 0) x = 0;
if (y + m_toolbar->height() > height()) y = selRect.top() - m_toolbar->height() - 10;
m_toolbar->move(x, y);

}

AnnotationEngine::AnnotationEngine(QObject *parent) : QObject(parent) , m_currentTool(None) , m_color(Qt::red) , m_penWidth(3) , m_isDrawing(false) { }

void AnnotationEngine::setCurrentTool(Tool tool) { m_currentTool = tool; }

void AnnotationEngine::setColor(const QColor &color) { m_color = color; }

void AnnotationEngine::setPenWidth(int width) { m_penWidth = qBound(1, width, 20); }

void AnnotationEngine::beginDraw(const QPoint &pos) { if (m_currentTool == None) return;

m_isDrawing = true;
m_currentAnnotation = Annotation();
m_currentAnnotation.tool = m_currentTool;
m_currentAnnotation.color = m_color;
m_currentAnnotation.penWidth = m_penWidth;
m_currentAnnotation.points.append(pos);
if (m_currentTool == Highlighter) {
    m_currentAnnotation.color = QColor(255, 255, 0, 100); // Yellow highlighter
}
}

void AnnotationEngine::continueDraw(const QPoint &pos) { if (!m_isDrawing || m_currentTool == None) return;

m_currentAnnotation.points.append(pos);
}

void AnnotationEngine::endDraw(const QPoint &pos) { if (!m_isDrawing || m_currentTool == None) return;

m_currentAnnotation.points.append(pos);
// Calculate bounding rect
if (!m_currentAnnotation.points.isEmpty()) {
    int minX = INT_MAX, minY = INT_MAX;
    int maxX = INT_MIN, maxY = INT_MIN;
    
    for (const QPoint &p : m_currentAnnotation.points) {
        minX = qMin(minX, p.x());
        minY = qMin(minY, p.y());
        maxX = qMax(maxX, p.x());
        maxY = qMax(maxY, p.y());
    }
    
    m_currentAnnotation.boundingRect = QRect(minX, minY, maxX - minX, maxY - minY);
}
// Clear redo stack when new drawing is added
m_redoStack.clear();
m_annotations.append(m_currentAnnotation);
m_isDrawing = false;
emit annotationAdded();
}

void AnnotationEngine::render(QPainter *painter, const QPoint &offset) { // Draw completed annotations for (const Annotation &annotation : m_annotations) { drawAnnotation(painter, annotation, offset); }

// Draw current annotation (while drawing)
if (m_isDrawing) {
    drawAnnotation(painter, m_currentAnnotation, offset);
}
}

void AnnotationEngine::drawAnnotation(QPainter *painter, const Annotation &annotation, const QPoint &offset) { if (annotation.points.isEmpty()) return;

painter->save();
switch (annotation.tool) {
    case Pen: {
        QPen pen(annotation.color, annotation.penWidth, Qt::SolidLine, Qt::RoundCap, Qt::RoundJoin);
        painter->setPen(pen);
        painter->setBrush(Qt::NoBrush);
        
        if (annotation.points.size() > 1) {
            QPainterPath path;
            path.moveTo(annotation.points.first() + offset);
            for (int i = 1; i < annotation.points.size(); ++i) {
                path.lineTo(annotation.points[i] + offset);
            }
            painter->drawPath(path);
        }
        break;
    }
    
    case Arrow: {
        if (annotation.points.size() < 2) break;
        
        QPoint start = annotation.points.first() + offset;
        QPoint end = annotation.points.last() + offset;
        
        QPen pen(annotation.color, annotation.penWidth, Qt::SolidLine, Qt::RoundCap);
        painter->setPen(pen);
        painter->drawLine(start, end);
        
        // Draw arrowhead
        double angle = qAtan2(end.y() - start.y(), end.x() - start.x());
        int arrowSize = 15;
        
        QPointF arrowP1 = end - QPointF(qCos(angle - M_PI / 6) * arrowSize,
                                        qSin(angle - M_PI / 6) * arrowSize);
        QPointF arrowP2 = end - QPointF(qCos(angle + M_PI / 6) * arrowSize,
                                        qSin(angle + M_PI / 6) * arrowSize);
        
        painter->setBrush(annotation.color);
        QPolygonF arrowHead;
        arrowHead << end << arrowP1 << arrowP2;
        painter->drawPolygon(arrowHead);
        break;
    }
    
    case Rectangle: {
        if (annotation.points.size() < 2) break;
        
        QPoint start = annotation.points.first() + offset;
        QPoint end = annotation.points.last() + offset;
        QRect rect(start, end);
        
        QPen pen(annotation.color, annotation.penWidth);
        painter->setPen(pen);
        painter->setBrush(Qt::NoBrush);
        painter->drawRect(rect.normalized());
        break;
    }
    
    case Highlighter: {
        QPen pen(annotation.color, annotation.penWidth * 5, Qt::SolidLine, Qt::FlatCap, Qt::BevelJoin);
        painter->setPen(pen);
        painter->setOpacity(0.4);
        
        if (annotation.points.size() > 1) {
            QPainterPath path;
            path.moveTo(annotation.points.first() + offset);
            for (int i = 1; i < annotation.points.size(); ++i) {
                path.lineTo(annotation.points[i] + offset);
            }
            painter->drawPath(path);
        }
        break;
    }
    
    case Blur: {
        if (annotation.points.size() < 2) break;
        
        QPoint start = annotation.points.first();
        QPoint end = annotation.points.last();
        QRect rect = QRect(start, end).normalized();
        
        drawBlurEffect(painter, rect, offset);
        break;
    }
    
    case Text: {
        if (annotation.text.isEmpty() || annotation.points.isEmpty()) break;
        
        painter->setPen(annotation.color);
        QFont font = painter->font();
        font.setPointSize(annotation.penWidth * 5 + 10);
        painter->setFont(font);
        
        painter->drawText(annotation.points.first() + offset, annotation.text);
        break;
    }
    
    case None:
    default:
        break;
}
painter->restore();
}

void AnnotationEngine::drawBlurEffect(QPainter *painter, const QRect &rect, const QPoint &offset) { // Simple pixelation effect for blur QRect targetRect = rect.translated(offset);

painter->save();
// Get the image from the painter's device
QImage *image = dynamic_cast<QImage*>(painter->device());
if (image) {
    QImage region = image->copy(targetRect);
    
    // Pixelate by scaling down and up
    int pixelSize = 8;
    QImage pixelated = region.scaled(
        qMax(1, region.width() / pixelSize),
        qMax(1, region.height() / pixelSize),
        Qt::IgnoreAspectRatio,
        Qt::FastTransformation
    ).scaled(
        region.width(),
        region.height(),
        Qt::IgnoreAspectRatio,
        Qt::FastTransformation
    );
    
    painter->drawImage(targetRect.topLeft(), pixelated);
} else {
    // Fallback: draw a semi-transparent rectangle
    painter->fillRect(targetRect, QColor(128, 128, 128, 180));
}
painter->restore();
}

void AnnotationEngine::addTextAnnotation(const QPoint &pos, const QString &text) { if (text.isEmpty()) return;

Annotation ann;

void AnnotationToolbar::onToolButtonClicked() { QPushButton button = qobject_cast<QPushButton>(sender()); if (!button) return;

int toolId = button->property("toolId").toInt();
// Toggle logic: if clicking same tool, deselect it
if (m_currentToolId == toolId) {
    toolId = AnnotationEngine::None;
}
// Update UI state
for (auto it = m_toolButtons.begin(); it != m_toolButtons.end(); ++it) {
    bool isSelected = (it.key() == toolId);
    it.value()->setProperty("selected", isSelected ? "true" : "false");
    it.value()->style()->unpolish(it.value());
    it.value()->style()->polish(it.value());
}
m_currentToolId = toolId;
emit toolSelected(toolId);
}

#include "HotkeyManager.h" #include #include

HotkeyManager& HotkeyManager::instance() { static HotkeyManager instance; return instance; }

HotkeyManager::HotkeyManager(QObject *parent) : QObject(parent) { // Install native event filter qApp->installNativeEventFilter(this);

// Register PrtSc hotkey (VK_SNAPSHOT = 0x2C)
if (registerHotkey(HOTKEY_CAPTURE, 0, VK_SNAPSHOT)) {
    qDebug() << "[HotkeyManager] PrtSc hotkey registered successfully";
} else {
    qWarning() << "[HotkeyManager] Failed to register PrtSc hotkey";
}
}

HotkeyManager::~HotkeyManager() { unregisterAllHotkeys(); qApp->removeNativeEventFilter(this); }

bool HotkeyManager::registerHotkey(int id, UINT modifiers, UINT virtualKey) { if (RegisterHotKey(nullptr, id, modifiers, virtualKey)) { m_registeredHotkeys.append(id); return true; } return false; }

void HotkeyManager::unregisterHotkey(int id) { UnregisterHotKey(nullptr, id); m_registeredHotkeys.removeAll(id); }

void HotkeyManager::unregisterAllHotkeys() { for (int id : m_registeredHotkeys) { UnregisterHotKey(nullptr, id); } m_registeredHotkeys.clear(); }

bool HotkeyManager::nativeEventFilter(const QByteArray &eventType, void *message, qintptr *result) { Q_UNUSED(result);

if (eventType == "windows_generic_MSG" || eventType == "windows_dispatcher_MSG") {
    MSG *msg = static_cast<MSG*>(message);
    
    if (msg->message == WM_HOTKEY) {
        int hotkeyId = static_cast<int>(msg->wParam);
        emit hotkeyTriggered(hotkeyId);
        
        if (hotkeyId == HOTKEY_CAPTURE) {
            emit captureRequested();
        }
        return true;
    }
}
return false;
}
